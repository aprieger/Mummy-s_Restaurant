CREATE TABLE CUSTOMER_ACCOUNT (
CUSTOMER_ID INT NOT NULL PRIMARY KEY,
FIRST_NAME VARCHAR(255),
LAST_NAME VARCHAR(255),
STREET VARCHAR(255),
CITY VARCHAR(255),
AREA_CODE INT,
PHONE_NUMBER INT,
USERNAME VARCHAR(255),
PASSWORD INT
);


CREATE TABLE CREDIT_CARDS (
CREDIT_ID INT NOT NULL PRIMARY KEY,
CUSTOMER_ID INT REFERENCES CUSTOMER_ACCOUNT (CUSTOMER_ID),
CARD_NUMBER INT,
BRAND VARCHAR(255),
SECURITY_NUMBER INT,
EXPERATION_DATE DATE,
NAME_ON_CARD VARCHAR(255),
STREET VARCHAR(255),
CITY VARCHAR(255),
AREA_CODE INT
);


CREATE TABLE ORDERS (
ORDER_ID INT NOT NULL PRIMARY KEY,
CUSTOMER_ID INT REFERENCES CUSTOMER_ACCOUNT (CUSTOMER_ID),
CREDIT_ID INT REFERENCES CREDIT_CARDS (CREDIT_ID),
PACKAGE_ID INT REFERENCES PACKAGES (PACKAGE_ID),
PAYMENT_TYPE INT,
PRICE NUMERIC(20,2),
QUANTITY INT,
STREET VARCHAR(255),
CITY VARCHAR(255),
AREA_CODE INT,
DELIVERY_DATE TIMESTAMP,
ORDER_DATE TIMESTAMP
);


CREATE TABLE PACKAGES (
PACKAGE_ID INT NOT NULL PRIMARY KEY,
NAME VARCHAR(255),
DESCRIPTION VARCHAR(255),
MEAL_CATEGORY INT,
IMAGE VARCHAR(255),
PACKAGE_PRICE INT,
IS_SPECIAL INT,
MEAL_TYPE INT,
PRICE NUMERIC(20,2)
);


CREATE TABLE SERVICE_AERA (
sERVICE_ID INT NOT NULL PRIMARY KEY,
PACKAGE_ID INT REFERENCES PACKAGES(PACKAGE_ID),
AREA_CODE INT
);


CREATE TABLE WORKER_ACCOUNTS (
EMPLOYEE_ID INT NOT NULL PRIMARY KEY,
USER_ID VARCHAR(255),
FIRST_NAME VARCHAR(255),
lAST_NAME VARCHAR(255),
PASSWORD VARCHAR(255),
PHONE_NUMBER INT,
IS_ADMIN INT
);
